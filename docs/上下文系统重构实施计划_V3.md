# 上下文系统重构实施计划 (V3.0)

> 基于《视效 Agent 闭环上下文与状态机重构设计方案 (V2.0)》，弃用 DSL，改为自然语言结构化描述。

---

## 一、核心改动概述

| 改动类型 | 原方案 | 新方案 |
|----------|--------|--------|
| **中间表示** | DSL AST (operators/topology) | 自然语言分层结构化描述 |
| **状态管理** | 单层 PipelineState | 四区划分（baseline/snapshot/gradient_window/checkpoint） |
| **历史注入** | 完整 shader 代码 | 梯度元数据（score/feedback_summary） |
| **回滚机制** | 逻辑标记 passed=False | 物理覆盖 current_shader = best_shader |
| **Inspect 反馈** | parameter_tuning 参数调整 | 语义描述（visual_issues/visual_goals） |
| **控制流** | Generate 循环 | 新增 Re-decompose 分支 |

## 二、状态总线四区划分

### 新 state.py 结构

```python
class PipelineState(TypedDict, total=False):
    # === 1. 只读基线区 (Read-Only Baseline) ===
    baseline: dict  # 单次任务不可变
    # {
    #   input_type: str,
    #   video_path: str | None,
    #   image_paths: list[str],
    #   user_notes: str,
    #   constraints: dict {max_alu, target_fps}
    # }
    
    # === 2. 当前快照区 (Current Snapshot) ===
    snapshot: dict  # 每轮更新
    # {
    #   visual_description: dict,  # Decompose 输出（自然语言）
    #   shader: str,               # 当前 GLSL 代码
    #   render_screenshots: list[str],
    #   inspect_feedback: dict,    # Inspect 输出
    #   iteration: int
    # }
    
    # === 3. 梯度记忆窗口 (Sliding Window History) ===
    gradient_window: list[dict]  # 最大长度 N (可配置)
    # [
    #   {
    #     iteration: int,
    #     score: float,
    #     feedback_summary: str,
    #     shader_diff_summary: str,  # 本轮修改了什么
    #     issues_fixed: list[str],   # 解决了哪些问题
    #     duration_ms: int
    #   }
    # ]
    
    # === 4. 回滚锚点区 (Checkpointing) ===
    checkpoint: dict  # 物理隔离备份
    # {
    #   best_score: float,
    #   best_shader: str,
    #   best_iteration: int,
    #   best_visual_description: dict
    # }
    
    # === 配置参数 (可调) ===
    config: dict
    # {
    #   max_iterations: int,
    #   passing_threshold: float,
    #   re_decompose_threshold: float,    # 触发重构阈值
    #   gradient_window_size: int,        # 梯度窗口长度 (默认 3)
    #   stagnation_variance: float        # 停滞判定波动阈值 (默认 0.05)
    # }
    
    # === Pipeline 元数据 ===
    pipeline_id: str
    status: str  # "running" | "passed" | "failed" | "re_decompose" | "max_iterations"
    error: str | None
    detailed_logs: list[PhaseLog]
    
    # === 用户人工干预 (保留) ===
    human_feedback: str | None
    human_iteration_mode: bool
    human_iteration_count: int
```

### 迁移策略

- 旧字段映射到新四区
- Pipeline 初始化时创建四区
- Agent 输出写入对应区域

## 三、上下文装配器

### 新建 context_assembler.py

为每个 Agent 组装专属上下文视图，严格隔离：

```python
class ContextAssembler:
    """Agent 上下文装配器：按文档要求组装专属视图"""
    
    @staticmethod
    def assemble_decompose(state: PipelineState, mode: str = "cold_start") -> dict:
        """
        Decompose Agent 上下文装配
        
        mode: "cold_start" | "re_decompose"
        """
        context = {
            "system_prompt": load_prompt("decompose_system.md"),
            "skill_context": SkillLoader.build_decompose_context(),
            "ux_reference": state["baseline"]["image_paths"],
        }
        
        if mode == "re_decompose":
            # 注入失败日志
            context["failure_log"] = build_failure_log(state)
        
        return context
    
    @staticmethod
    def assemble_generate(state: PipelineState) -> dict:
        """
        Generate Agent 上下文装配
        
        禁止注入：原始图片、渲染截图
        """
        context = {
            "system_prompt": load_prompt("generate_system.md"),
            "skill_context": SkillLoader.build_generate_context(),
            "visual_description": state["snapshot"]["visual_description"],
            "current_shader": state["snapshot"]["shader"],
            "inspect_feedback": state["snapshot"]["inspect_feedback"],
            "gradient_window": truncate_gradient_window(state["gradient_window"]),
        }
        
        if state.get("human_feedback"):
            context["human_feedback"] = state["human_feedback"]
        
        return context
    
    @staticmethod
    def assemble_inspect(state: PipelineState) -> dict:
        """
        Inspect Agent 上下文装配
        
        禁止注入：DSL、参数约束
        注入：完整 shader（不截断）
        """
        context = {
            "system_prompt": load_prompt("inspect_system.md"),
            "skill_context": SkillLoader.build_inspect_context(),
            "ux_reference": state["baseline"]["image_paths"],
            "render_screenshots": state["snapshot"]["render_screenshots"],
            "shader": state["snapshot"]["shader"],  # 完整代码
            "gradient_window": truncate_gradient_window(state["gradient_window"]),
        }
        
        return context
```

## 六、物理回滚机制

### graph.py node_inspect 改动

```python
# Inspect 发现劣化时，物理覆盖 shader
checkpoint = state.get("checkpoint", {})
best_score = checkpoint.get("best_score", 0)
current_score = result.get("overall_score", 0)

if current_score < best_score:
    # === 物理回滚 ===
    snapshot = state.get("snapshot", {})
    snapshot["shader"] = checkpoint["best_shader"]
    snapshot["iteration"] = checkpoint["best_iteration"]
    snapshot["visual_description"] = checkpoint.get("best_visual_description", {})
    
    # === 强注入指令 ===
    rollback_instruction = f"""
[系统回滚指令]
本次修改导致得分从 {best_score:.2f} 跌至 {current_score:.2f}。
系统已在底层回滚到第 {checkpoint['best_iteration']} 轮优质代码。

请废弃刚才的修改方向，基于当前恢复的代码探索新效果。
避免重复之前的错误方向。
"""
    result["visual_issues"].append(rollback_instruction)
    
    print(f"[Inspect Node] Score regression detected, rolling back to iteration {checkpoint['best_iteration']}")
```

## 七、梯度裁剪机制

### truncate_gradient_window 函数

```python
def truncate_gradient_window(gradient_window: list[dict], max_size: int = 3) -> list[dict]:
    """
    梯度记忆窗口裁剪
    
    仅保留最近 N 轮的梯度元数据，禁止存放完整代码。
    """
    if not gradient_window:
        return []
    
    # 截取最近 N 轮
    truncated = gradient_window[-max_size:]
    
    # 确保不包含完整 shader（仅元数据）
    for entry in truncated:
        # 删除可能存在的 shader 字段
        entry.pop("shader", None)
        entry.pop("previous_shader", None)
    
    return truncated
```

## 八、Re-decompose 路由

### 触发条件

```python
def should_trigger_re_decompose(state: PipelineState) -> bool:
    """
    判断是否触发拓扑重构
    
    条件：
    1. current_score < re_decompose_threshold (可配置)
    2. 或连续 N 轮停滞 (波动 < stagnation_variance)
    """
    config = state.get("config", {})
    snapshot = state.get("snapshot", {})
    gradient_window = state.get("gradient_window", [])
    
    # 条件 1：低于阈值
    current_score = snapshot.get("inspect_feedback", {}).get("overall_score", 0)
    threshold = config.get("re_decompose_threshold", 0.5)
    if current_score < threshold:
        return True
    
    # 条件 2：停滞检测
    window_size = config.get("gradient_window_size", 3)
    variance_threshold = config.get("stagnation_variance", 0.05)
    
    if len(gradient_window) >= window_size:
        recent_scores = [e.get("score", 0) for e in gradient_window[-window_size:]]
        variance = max(recent_scores) - min(recent_scores)
        if variance < variance_threshold:
            return True
    
    return False
```

### 控制流路由

```python
# graph.py route_from_inspect
def route_from_inspect(state: PipelineState) -> Literal["generate", "decompose", "end"]:
    if state.get("error"):
        return "end"
    
    if state.get("passed"):
        return "end"
    
    # === Re-decompose 分支 ===
    if should_trigger_re_decompose(state):
        return "decompose"  # 触发重构
    
    # 正常迭代
    if state.get("iteration", 0) >= state.get("config", {}).get("max_iterations", 5):
        return "end"
    
    return "generate"
```

### Failure Log 构建

```python
def build_failure_log(state: PipelineState) -> str:
    """
    构建 Re-decompose 时注入 Decompose 的失败日志
    """
    snapshot = state.get("snapshot", {})
    gradient_window = state.get("gradient_window", [])
    
    # 提取失败信息
    last_feedback = snapshot.get("inspect_feedback", {})
    visual_issues = last_feedback.get("visual_issues", [])
    
    # 构建负样本
    failure_log = f"""
=== 重构阻断触发 ===

前一版拓扑未能收敛，失败原因：
{chr(10).join(visual_issues[:5])}

近 {len(gradient_window)} 轮评分：{[e.get('score') for e in gradient_window[-3:]]}

建议：更换底层算子或拓扑结构，避免重复之前的失败方向。
"""
    return failure_log
```
